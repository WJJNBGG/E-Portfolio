Example R code: NDVI with simple raster layers

library(terra)
Warning: package 'terra' was built under R version 4.5.1
terra 1.8.74
red <- rast(nrows = 4, ncols = 4, vals = c(
  0.2, 0.3, 0.4, 0.2,
  0.1, 0.2, 0.3, 0.2,
  0.2, 0.4, 0.5, 0.3,
  0.1, 0.2, 0.2, 0.1
))

nir <- rast(nrows = 4, ncols = 4, vals = c(
  0.6, 0.7, 0.8, 0.6,
  0.5, 0.6, 0.7, 0.5,
  0.6, 0.8, 0.9, 0.7,
  0.4, 0.5, 0.6, 0.4
))

ndvi <- (nir - red) / (nir + red)
plot(ndvi, main = "Example NDVI")

Example R code: comparing spectral signatures

library(ggplot2)
Warning: package 'ggplot2' was built under R version 4.5.1
spectra <- data.frame(
  wavelength = c(450, 550, 650, 850, 1600, 2200),
  vegetation = c(0.08, 0.12, 0.05, 0.52, 0.30, 0.18),
  soil = c(0.15, 0.20, 0.25, 0.30, 0.35, 0.32)
)

spectra_long <- reshape(
  spectra,
  varying = c("vegetation", "soil"),
  v.names = "reflectance",
  timevar = "class",
  times = c("Vegetation", "Soil"),
  direction = "long"
)

ggplot(spectra_long, aes(x = wavelength, y = reflectance, color = class)) +
  geom_line(linewidth = 1) +
  geom_point(size = 2) +
  labs(title = "Example Spectral Profiles",
       x = "Wavelength (nm)",
       y = "Reflectance")
